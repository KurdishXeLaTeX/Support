\section{پێشەکی}
\end{flushright}
یاسای ساینەکان، لە ئەندازە بە کەڵکە بۆ شیکاری پرسیارەکانی وەک: دۆزینەوەی لاکانی سێگۆشە یان گۆشەکانی سێگۆشە.
\end{center}
\textbf{پێناسە:} لە سێگۆشە زانییدا، \textbf{یاسای ساینەکان} بریتییە لە هاوکێشەیەک، کە پەیوەندی درێژی لایەکانی سێگۆشەیەک بە گۆشەکانی سێگۆشەکە دەبەستێتەوە.\\

\textbf{{\large دوو شێوازی هەیە}}.\\

\textbf{یەکەم:} بۆ دۆزینەوەی درێژی لایەکی نەزانراوی سێگۆشەکە.\\

\textbf{دووەم:} بۆ دۆزینەوەی گۆشەی لایەکی نەزانراوی سێگۆشەکە.\\


\textbf{هاوکێشەی یەکەم}:$$\frac{a}{\sin \alpha}=\frac{b}{\sin \beta}=\frac{c}{\sin \gamma}$$\\


\textbf{هاوکێشەی دووەم}:$$\frac{\sin \alpha}{a}=\frac{\sin \beta}{b}=\frac{\sin \gamma}{c}$$
 هەریەک لە $a$، $b$ وە $c$ بریتیین لە درێژی لای سێگۆشەکە. وە $\alpha$، $\beta$ لەگەڵ $\gamma$ بریتیین لە گۆشەکانی
 سێگۆشەکە.\\
 
 
 بەڵام هاوکێشەی یەکەم قسە دەکات! دەڵێ: "ئەگەر لای $a$ دابەشی گۆشەی بەرامبەری خۆی $(\sin \alpha)$ بکرێت، ئەوا بۆ هەموو لایەکانی تریش ڕاستەو بەمەش نرخی درێژی هەر لایەک  دەدۆزرێتەوەو یەکسان دەبێت بە ئەوی تر وەکو زنجیرە بە دووایەک! $b$ بۆ $\sin \beta$، $c$ بۆ $\sin \gamma$، هتد...".\\
هاوکێشەی دووەمیش قسە دەکات! دەڵێ: "ئەگەر هاوکێشەی یەکەم هەڵبگەڕێنینەوە، ئەوا گۆشەی نەزانراوی لایەک دەدۆزرێتەوەو یەکسانە بە گۆشەکانی لایەکانی تر، یەک بە دووایەک".

\subsection{سەلماندن}
\begin{center}
لێرە سەلماندن پەیڕەو دەکەین، تا هاوکێشەی یەکەمی بنەڕەتی دەربکەین.\\
\end{center}
 سەیری ئەم وێنەیە بکەو بیسەلمێنە کە $\frac{a}{\sin \alpha}=\frac{b}{\sin \beta}=\frac{c}{\sin \gamma}$ لە کاتێک، سێگۆشەیەکی ئاسایی پارچەکراوە بۆ دووبەش بە بەرزییەک، کە ناوی $h$ وە هەریەک لە $a$ و $b$ و $c$ دەبنە درێژی لایەکانی سێگۆشە پارچەکراوەکە و هەر یەک لە $\alpha$ و $\beta$ و $\gamma$ دەبنە گۆشەکانی سێگۆشەی پارچەکراو. بە زانینی ئەوەی نەخشەی \textbf{ساین} بریتییە لە "بەرامبەر لەسەر ژێ". هەرئێستا ئەمە لەسەر هەردوو بەشی سێگۆشەکە جێبەجێ بکە.\\
 
 
 
\textbf{کاتێک}: $h$ دەبێتە \textbf{بەرامبەر} بۆ هەردوولا وە هەریەک لە $a$ و $b$ دەبنە \textbf{ژێ}.\\


\begin{flushleft}

\draw (C) -- node[above] {$\hspace{-0.3cm} b$} (B) -- node[right] {$\hspace{0.3cm}  a$} (A) -- node[below] {$c$} (C);
\pic [my angle, "$\beta$"] {angle=B--A--C};
\pic [my angle, "$\alpha$"] {angle=A--C--B};
\pic [my angle, "$\hspace{0.3cm} \gamma$"] {angle=C--B--A};

\end{tikzpicture}\\
\end{flushleft}




بەمەش: $$\Rightarrow \sin \alpha=\frac{h}{b} \Rightarrow b \sin \alpha=h$$  
$$\Rightarrow \sin \beta=\frac{h}{a} \Rightarrow a \sin \beta=h$$


کەواتە: $$\Rightarrow a \sin \beta=b \sin \alpha$$


دووای ڕێکخستن: $$ \hspace{0.5cm}\therefore \frac{a}{\sin \alpha}=\frac{b}{\sin \beta}$$


\textbf{تێبینی:} دەتووانین هەمان ڕێگا بەکاربهێنین بۆ دۆزینەوەو سەلماندنی $\frac{c}{\sin \gamma}$. بەڵام لێرە زۆر درێژەی


 پێ نادەین لەسەر بابەتەکە. پێم خۆشە خوێنەر خۆی سەلماندنی بۆ بکات...

\section{شیکارکردن}
لێرە دەست دەکەین بە شیکاری نموونە، تا تێگەیشتن لێمان نزیک بێت و نەزانی لێمان دوور بێت.
\subsection{شیکاری نموونەیەک لەسەر هاوکێشەی یەکەم}
\textbf{نموونەکە}: وێنەی خوارەوە بەکاربهێنە بۆ دۆزینەوی درێژی $x$، نرخی $x$ چەندە؟ کارەکانت  ڕوونبکەوە.

\pic [my angle, "$35\degree$"] {angle=C--B--A};
\pic [my angle, "$105\degree$"] {angle=A--C--B};
\end{tikzpicture}\\


\textbf{شیکار}: $$\Rightarrow \frac{b}{\sin \beta}=\frac{7}{\sin 35\degree}=\frac{c}{\sin \gamma}=\frac{x}{\sin 105\degree}$$
$$\Rightarrow \frac{x}{\sin 105\degree}=\frac{7}{\sin 35\degree}$$
$$\Rightarrow x \sin 35\degree=7 \sin 105\degree$$
$$\therefore x=11.8$$

نموونەکە: وێنەی خوارەوە بەکاربهێنە بۆ دۆزینەوی گۆشەی $\gamma$، نرخی $\gamma$ چەندە؟ کارەکانت  ڕوونبکەوە.

\textbf{شیکار:} $$\Rightarrow \frac{\sin \alpha}{a}=\frac{\sin 63\degree}{5.5}=\frac{\sin \gamma}{c}=\frac{\sin \gamma}{4.7}$$
$$\Rightarrow \frac{\sin \gamma}{4.7}=\frac{\sin 63\degree}{5.5}$$
$$\Rightarrow 5.5 \sin \gamma=4.7 \sin 63\degree$$
$$\Rightarrow \sin \gamma=0.7614\ldots$$
$$\Rightarrow \gamma=\arcsin \hspace{0.1cm} (0.7614\ldots)$$
$$\Rightarrow \gamma=49.6\degree$$
